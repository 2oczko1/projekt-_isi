        super(name, email, phone, "Admin", password);
    }

    // Dodatkowe metody

        System.out.println("System user added: " + user.getName());
    }

    public void manageAccessRights(AppUser user) {
        System.out.println("Access rights managed for user: " + user.getName());
    }
}
